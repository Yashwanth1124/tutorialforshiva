#include <stdio.h>
#include <limits.h>  // For INT_MAX

#define MAX_VERTICES 20

// Define the structure with appropriate members
struct full {
    char dest[10];
    char line[10];
    int hops;
} f[MAX_VERTICES];

int main() {
    int nv, min, minver, i;
    char sv[10], temp;

    // Input number of vertices
    printf("Enter number of vertices: ");
    scanf("%d", &nv);

    // Input source vertex
    printf("Enter source vertex: ");
    scanf("%s", sv);

    // Input the full table for source vertex
    printf("Enter full table for source vertex %s:\n", sv);
    for (i = 0; i < nv; i++) {
        scanf("%s %s %d", f[i].dest, f[i].line, &f[i].hops);
    }

    // Print hierarchical table
    printf("\nHIERARCHICAL TABLE\n\n");

    for (i = 0; i < nv;) {
        if (sv[0] == f[i].dest[0]) {
            printf("\n%s %s %d", f[i].dest, f[i].line, f[i].hops);
            i++;
        } else {
            min = INT_MAX;
            minver = 0;
            temp = f[i].dest[0];
            while (i < nv && temp == f[i].dest[0]) {
                if (min > f[i].hops) {
                    min = f[i].hops;
                    minver = i;
                }
                i++;
            }
            printf("\n%c %s %d", temp, f[minver].line, f[minver].hops);
        }
    }

    return 0;
}
