#include <stdio.h>
#include <stdlib.h> // for exit()
#include <string.h>

int main()
{
    int i, j, k, l, n, count;
    char s[100], cs[50];

    printf("\n ENTER THE STUFFED STRING :");
    fgets(s, sizeof(s), stdin); // Use fgets instead of gets
    n = strlen(s);

    printf("\nTHE STRING IS\n");

    for (i = 0; i < n;)
    {
        if (s[i] == '$')
        {
            i++;
            count = (s[i] - '0') * 10 + (s[i + 1] - '0');
            if (count < 5)
            {
                printf("INVALID MESSAGE\n");
                return 1; // Use return instead of exit
            }
            while (count > 0)
            {
                printf("%c", s[i + 2]);
                count--;
            }
            i = i + 3;
        }
        else
        {
            printf("%c", s[i]);
            i++;
        }
    }

    return 0; // main function should return an integer
}
